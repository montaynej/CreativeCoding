# Finished Code

```js {33-53}{lines:true,startLine:1, maxHeight:'440px'}

let cols = 10;
let rows = 10;
let cellSize = 40;
let gridWidth = cols * cellSize;
let gridHeight = rows * cellSize;
let counter = 0;
let sizes = [];

let colors = ["#ff0000","#ff7f00","#ffff00","#7fff00","#00ff00","#00ff7f","#00ffff","#007fff","#0000ff","#8b00ff",];

function setup() {
	createCanvas(500, 500);
	noStroke();
	angleMode(RADIANS);
	rectMode(CENTER);
	for (let i = 0; i < 100; i++) {
		sizes.push(Math.floor(random(10, 40)));
	}
}

function draw() {
	background(55, 0, 0);
	// centre the grid on the canvas
	let offsetX = (width - gridWidth) / 2 + cellSize / 2;
	let offsetY = (height - gridHeight) / 2 + cellSize / 2;

	push();
	translate(offsetX, offsetY);

	for (let y = 0; y < rows; y++) {
		for (let x = 0; x < cols; x++) {
			let cx = x * cellSize;
			let cy = y * cellSize;

			let index = (x + y) % colors.length;
			let rotation = random(TWO_PI);

			push();
			noFill();
			stroke(100);
			translate(cx, cy);
			rect(0, 0, cellSize);
			rotate(rotation);
			noStroke();
			fill(colors[index]);
			let scaler = map(sin(counter), -1, 1, 0.8, 1.5);
			let cell = y * 10 + x;
			scale(scaler);
			ellipse(0, 0, sizes[cell]);
			stroke(125);
			//line(0, 0, 30, 0);
			pop();
		}
	}
	pop();

	counter = counter + 0.07;
}
```
